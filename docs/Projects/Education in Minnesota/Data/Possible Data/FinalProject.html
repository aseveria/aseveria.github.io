pSchools <- read_xlsx("PublicSchool.xlsx",range=cell_cols("C:Y"))
states_map <- map_data("state")



newS<-pSchools%>%
    mutate(state_name = str_to_lower(abbr2state(`STATE`)))%>%
      filter(SCHOOLYEAR == "2021-2022")%>%
    count(`state_name`) 



newS %>%
  ggplot() +
  geom_map(
    map = states_map,
    aes( map_id = state_name,fill = n)) +
  expand_limits(x = states_map$long, y = states_map$lat) +
  scale_fill_viridis()+

  theme_map()+
